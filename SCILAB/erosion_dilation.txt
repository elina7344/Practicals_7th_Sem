clc;
clear;
close;
I=imread('E:\SEM 7\IWUS\err_dil.jpg');
subplot(2,3,1);
imshow(I);
title("original image");
se=imcreatese('ellipse',7,7);
dilate=imdilate(I,se);
subplot(2,3,2);
imshow(dilate);
title("Dilated image");
erode=imerode(I,se);
subplot(2,3,3);
imshow(erode);
title("Eroded Image");
open=imopen(I,se);
subplot(2,3,4);
imshow(open);
title("opened image");
close=imclose(I,se);
subplot(2,3,5);
imshow(close);
title("Close image");
